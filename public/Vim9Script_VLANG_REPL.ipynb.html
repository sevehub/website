    f.write("""
vim9script
def VimNew(): number
  var sum = 0
  for i in range(1, 29999)
      sum += i
  endfor
  return sum
enddef

var start = reltime()
echo VimNew()
echo 'Vim new: ' .. reltimestr(reltime(start))
    """)
