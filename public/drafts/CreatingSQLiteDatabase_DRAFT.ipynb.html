Example Jupyter Notebook: Creating a SQLite Database¶

This notebook demonstrates how to create a SQLite database from data defined in the cells.

In [1]:
# Cell 1: Import necessary libraries
import sqlite3
import pandas as pd
In [2]:
# Cell 2: Define some sample data for PC items
data = {
    'item_name': ['Graphics Card', 'Motherboard', 'RAM', 'SSD', 'Power Supply'],
    'quantity': [10, 5, 20, 15, 8],
    'price': [499.99, 149.99, 79.99, 89.99, 59.99]
}

# Create a DataFrame
df = pd.DataFrame(data)
df

In [3]:
# Cell 3: Connect to SQLite database (or create it if it doesn't exist)
conn = sqlite3.connect('data/example.db')

# Create a cursor object
cursor = conn.cursor()
In [4]:
# Cell 4: Create a table in the SQLite database
cursor.execute('''
CREATE TABLE IF NOT EXISTS pc_items (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    item_name TEXT NOT NULL,
    quantity INTEGER NOT NULL,
    price REAL NOT NULL
)
''')

# Commit the changes
conn.commit()
In [5]:
# Cell 5: Insert data into the table
for index, row in df.iterrows():
    cursor.execute('''
    INSERT INTO pc_items (item_name, quantity, price) VALUES (?, ?, ?)
    ''', (row['item_name'], row['quantity'], row['price']))

# Commit the changes
conn.commit()
In [6]:
# Cell 6: Query the database to verify the data was inserted
cursor.execute('SELECT * FROM pc_items')
rows = cursor.fetchall()

# Display the results
for row in rows:
    print(row)

# Cell 7: Close the database connection
conn.close()
