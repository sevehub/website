def display_classification_report(X, y, X_test, y_test, estimator):
    """
    yb classification report 
    """
    visualizer = ClassificationReport(estimator,support=True)

    visualizer.fit(X, y)  
    visualizer.score(X_test, y_test)  
    visualizer.show()
